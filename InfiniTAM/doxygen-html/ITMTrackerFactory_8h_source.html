145  static std::vector<TrackerIterationType> parseLevelConfig(const char *str)
146  {
147  bool parseError = false;
148  std::vector<TrackerIterationType> ret;
149  for (int i = static_cast<int>(strlen(str))-1; i >= 0; --i) {
150  switch (str[i]) {
151  case 'r':
152  ret.push_back(TRACKER_ITERATION_ROTATION);
153  break;
154  case 't':
155  ret.push_back(TRACKER_ITERATION_TRANSLATION);
156  break;
157  case 'b':
158  ret.push_back(TRACKER_ITERATION_BOTH);
159  break;
160  case 'n':
161  ret.push_back(TRACKER_ITERATION_NONE);
162  break;
163  default:
164  parseError = true;
165  break;
166  }
167  }
168 
169  if (parseError) {
170  fprintf(stderr, "error parsing level configuration '%s'\n", str);
171  for (int i = 0; (unsigned)i < ret.size(); ++i) fprintf(stderr, "level %i: %i\n", (int)ret.size()-i, (int)(ret[ret.size()-i]));
172  }
173  return ret;
174  }
