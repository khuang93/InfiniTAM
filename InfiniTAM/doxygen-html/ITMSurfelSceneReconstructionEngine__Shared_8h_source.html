42 inline float calculate_gaussian_sample_confidence(int locId, int width, int height, float sigma)
43 {
44  // Calculate the normalised radial distance of the depth sample from the camera centre.
45  const int x = locId % width, y = locId / width;
46  const float halfW = width / 2.0f, halfH = height / 2.0f;
47  const float dx = fabs(x - halfW), dy = fabs(y - halfH);
48  const float gamma = sqrtf((dx * dx + dy * dy) / (halfW * halfW + halfH * halfH));
49 
50  // Calculate and return the confidence value itself.
51  return expf(-gamma*gamma) / (2*sigma*sigma);
52 }

136 inline TSurfel merge_surfels(const TSurfel& target, const TSurfel& source, float maxSurfelRadius, bool shouldMergeProperties, RadiusCombinationMode radiusCombinationMode)
137 {
138  TSurfel result = target;
139  const float confidence = result.confidence + source.confidence;
140 
141  if(shouldMergeProperties)
142  {
143 #if DEBUG_CORRESPONDENCES
144  result.newPosition = source.position;
145  result.oldPosition = result.position;
146 #endif
147 
148  // Compute confidence-weighted averages of the positions and normals of the input surfels. Note that this can
149  // result in a combined normal that is no longer unit length, so we must renormalise it accordingly.
150  result.position = (result.confidence * result.position + source.confidence * source.position) / confidence;
151  result.normal = (result.confidence * result.normal + source.confidence * source.normal) / confidence;
152  result.normal /= length(result.normal);
153 
154  // Compute the radius of the combined surfel.
155  switch(radiusCombinationMode)
156  {
157  case RCM_DISTANCE:
158  result.radius = length(target.position - source.position);
159  break;
160  case RCM_CONFIDENCEWEIGHTEDAVERAGE:
161  default:
162  result.radius = (result.confidence * result.radius + source.confidence * source.radius) / confidence;
163  break;
164  }
165 
166  // Ensure that the radius of the combined surfel does not exceed the maximum permitted radius.
167  if(result.radius > maxSurfelRadius) result.radius = maxSurfelRadius;
168 
169  // Compute a confidence-weighted average of the colours of the input surfels.
170  Vector3u resultColour = SurfelColourManipulator<TSurfel::hasColourInformation>::read(result);
171  Vector3u sourceColour = SurfelColourManipulator<TSurfel::hasColourInformation>::read(source);
172  Vector3u colour = ((result.confidence * resultColour.toFloat() + source.confidence * sourceColour.toFloat()) / confidence).toUChar();
173  SurfelColourManipulator<TSurfel::hasColourInformation>::write(result, colour);
174  }
175 
176  result.confidence = confidence;
177  if(source.timestamp > result.timestamp) result.timestamp = source.timestamp;
178 
179  return result;
180 }

335 inline void calculate_radius(int locId, const float *depthMap, const Vector3f *normalMap, const ITMIntrinsics& intrinsics, float *radiusMap)
336 {
337  float r = 0.0f;
338  Vector3f n = normalMap[locId];
339 
340  if(length(n) > 0.0f)
341  {
342  // The intuition behind the radius calculation is that you want the surfel to fully cover a single pixel on the image plane (at distance f).
343  // To do this, it needs to have a projected radius of sqrt(2). Projecting it down onto the image plane from distance d means multiplying its
344  // radius by f / d, hence its unprojected radius should be sqrt(2) / (f / d) = sqrt(2) * d / f. Note that this is the radius calculation used
345  // in the "Dense Planar SLAM" paper, rather than the one used in the original Keller paper.
346  float d = depthMap[locId];
347  float f = 0.5f * (intrinsics.projectionParamsSimple.fx + intrinsics.projectionParamsSimple.fy);
348  r = sqrt(2.0f) * d / f;
349  }
350 
351  radiusMap[locId] = r;
352 }

364 inline void calculate_vertex_position(int locId, int width, const ITMIntrinsics& intrinsics, const float *depthMap, Vector4f *vertexMap)
365 {
366  /*
367  v(~u~) = D(~u~) K^{-1} (~u~^T,1)^T
368  = D(~u~) (fx 0 px)^{-1} (ux) = D(~u~) ((ux - px) / fx)
369  (0 fy py) (uy) ((uy - py) / fy)
370  (0 0 1) ( 1) ( 1)
371  */
372  int ux = locId % width, uy = locId / width;

382  depth,
383  1.0f
384  );
385  }

432 inline void find_corresponding_surfel(int locId, const Matrix4f& invT, const float *depthMap, int depthMapWidth, const Vector3f *normalMap, const unsigned int *indexImage,
433  int supersamplingFactor, const TSurfel *surfels, unsigned int *correspondenceMap, unsigned short *newPointsMask)
434 {
435  // If the depth pixel or normal is invalid, early out.
436  const float EPSILON = 1e-3f;
437  float depth = depthMap[locId];
438  if(fabs(depth + 1) <= EPSILON || length(normalMap[locId]) <= EPSILON)
439  {
440  correspondenceMap[locId] = 0;
441  newPointsMask[locId] = 0;
442  return;
443  }
444 
445  // Otherwise, find corresponding surfels in the scene and pick the best one (if any).
446  int bestSurfelIndex = -1;
447  float bestSurfelConfidence = 0.0f;
448  int ux = locId % depthMapWidth, uy = locId / depthMapWidth;
449  for(int dy = 0; dy < supersamplingFactor; ++dy)
450  {
451  for(int dx = 0; dx < supersamplingFactor; ++dx)
452  {
453  int x = ux * supersamplingFactor + dx;
454  int y = uy * supersamplingFactor + dy;
455  int surfelIndex = indexImage[y * depthMapWidth * supersamplingFactor + x] - 1;
456  if(surfelIndex >= 0)
457  {
458  // TODO: Make this slightly more sophisticated, as per the paper.
459  TSurfel surfel = surfels[surfelIndex];
460  Vector3f liveSurfelPos = transform_point(invT, surfel.position);
461  float surfelDepth = liveSurfelPos.z;
462 
463  const float deltaDepth = 0.01f;
464  if(surfel.confidence > bestSurfelConfidence && fabs(surfelDepth - depth) <= deltaDepth)
465  {
466  bestSurfelIndex = surfelIndex;
467  bestSurfelConfidence = surfel.confidence;
468  }
469  }
470  }
471  }
472 
473  // Record any corresponding surfel found, together with a flag indicating whether or not we need to add a new surfel.
474  correspondenceMap[locId] = bestSurfelIndex >= 0 ? bestSurfelIndex + 1 : 0;
475  newPointsMask[locId] = bestSurfelIndex >= 0 ? 0 : 1;
476 }

493 template <typename TSurfel>
494 _CPU_AND_GPU_CODE_
495 inline void find_mergeable_surfel(int locId, const unsigned int *indexImage, int indexImageWidth, int indexImageHeight, const unsigned int *correspondenceMap, const TSurfel *surfels,
496  float stableSurfelConfidence, float maxMergeDist, float maxMergeAngle, float minRadiusOverlapFactor, unsigned int *mergeTargetMap)
497 {
498  // Look up the surfel at the current location. If there isn't one, early out.
499  int surfelIndex = indexImage[locId] - 1;
500  if(surfelIndex == -1) return;
501  const TSurfel surfel = surfels[surfelIndex];
502 
503  // Determine whether the surfel itself can justify the merge (it needs to be stable and to have been updated this frame).
504  bool surfelCanJustify = surfel.confidence >= stableSurfelConfidence && correspondenceMap[locId] > 0;
505 
506  // For each neighbour of the current location that has a higher raster index:
507  int x = locId % indexImageWidth, y = locId / indexImageWidth;
508  int neighbourX[] = { x + 1, x - 1, x, x + 1 };
509  int neighbourY[] = { y, y + 1, y + 1, y + 1 };
510  int bestMergeSource = -1;
511 
512  for(int i = 0; i < 4; ++i)
513  {
514  // If the neighbour is out of range, continue.
515  if(neighbourX[i] >= indexImageWidth || neighbourY[i] >= indexImageHeight) continue;
516 
517  // Look up the surfel (if any) at the neighbour. If there isn't one, or it's the same as this surfel, continue.
518  int neighbourLocId = neighbourY[i] * indexImageWidth + neighbourX[i];
519  int neighbourSurfelIndex = indexImage[neighbourLocId] - 1;
520  if(neighbourSurfelIndex == -1 || neighbourSurfelIndex == surfelIndex) continue;
521  const TSurfel neighbourSurfel = surfels[neighbourSurfelIndex];
522 
523  // If the merge cannot be justified, continue.
524  if(!surfelCanJustify && !(neighbourSurfel.confidence >= stableSurfelConfidence && correspondenceMap[neighbourLocId] > 0)) continue;
525 
526  // If the difference in positions and the angle between the normals are sufficiently small, and the radii significantly overlap, update the best merge source.
527  float dist = length(surfel.position - neighbourSurfel.position);
528  float angle = acosf(dot(surfel.normal, neighbourSurfel.normal));
529  if(dist <= maxMergeDist && angle <= maxMergeAngle && dist * minRadiusOverlapFactor <= surfel.radius + neighbourSurfel.radius)
530  {
531  bestMergeSource = neighbourLocId;
532  }
533  }
534 
535  // If there was a best merge source, record the current location as its merge target.
536  if(bestMergeSource != -1) mergeTargetMap[bestMergeSource] = locId + 1;
537 }
