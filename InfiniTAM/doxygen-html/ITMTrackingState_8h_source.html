60 
61  Vector3f cameraCenter_pc = -1.0f * (pose_pointCloud->GetR().t() * pose_pointCloud->GetT());

64  Vector3f diff3 = cameraCenter_pc - cameraCenter_live;
65 
66  float diff = diff3.x * diff3.x + diff3.y * diff3.y + diff3.z * diff3.z;

68  // if the camera center has moved by more than a threshold

70
