9 _CPU_AND_GPU_CODE_ inline float tukey_rho(float r, float c)
10 {
11  const float c_sq_6 = c * c / 6.f;
12  if(fabs(r) <= c)
13  {
14  float tukey_r = r / c;
15  tukey_r *= tukey_r;
16  tukey_r = 1 - tukey_r;
17  tukey_r = tukey_r * tukey_r * tukey_r;
18 
19  return c_sq_6 * (1.f - tukey_r);
20  }
21  else
22  {
23  return c_sq_6;
24  }
25 }

172  if (x >= imgSize_depth.x || y >= imgSize_depth.y) return false;
173 
174  // Point in current camera coordinates
175  const Vector4f pt_curr = points_curr[y * imgSize_depth.x + x];
176  // no need to interpolate, has already been done in projectPoint_exRGB
177  const float intensity_curr = intensities_curr[y * imgSize_depth.x + x];
178 
179  // Invalid point or too far away
180  if (pt_curr.w < 0.f || intensity_curr < 0.f || pt_curr.z < 1e-3f || pt_curr.z > viewFrustum_max) return false;
181 
182  // Transform the point in world coordinates
183  const Vector3f pt_world = approxInvPose * pt_curr.toVector3();
184 
185  // Transform the point in previous camera coordinates
186  const Vector3f pt_prev = scenePose * pt_world;
187 
188  if (pt_prev.z <= 0.0f) return false; // Point behind the camera
189 
190  // Project the point in the previous intensity frame
191  const Vector2f pt_prev_proj = project(pt_prev, intrinsics_rgb);
192 
193  // Outside the image plane
194  if (pt_prev_proj.x < 0 || pt_prev_proj.x >= imgSize_rgb.x - 1 ||
195  pt_prev_proj.y < 0 || pt_prev_proj.y >= imgSize_rgb.y - 1) return false;
196 
197  // Point should be valid, sample intensities and gradients
198  const float intensity_prev = interpolateBilinear_single(intensities_prev, pt_prev_proj, imgSize_rgb);
199  const Vector2f gradient_prev = interpolateBilinear_Vector2(gradients, pt_prev_proj, imgSize_rgb);
200 
201  const float intensity_diff = intensity_prev - intensity_curr;
202 
203  if (fabs(intensity_diff) >= tukeyCutoff * colourThresh) return false; // Difference too big
204  if (fabs(gradient_prev.x) < minGradient || fabs(gradient_prev.y) < minGradient) return false; // Gradient too small
205 
206  // Cache rows of the scenePose rotation matrix, to be used in the pose derivative
207  const Vector3f scene_rot_row_0 = scenePose.getRow(0).toVector3();
208  const Vector3f scene_rot_row_1 = scenePose.getRow(1).toVector3();
209  const Vector3f scene_rot_row_2 = scenePose.getRow(2).toVector3();
210 
211  // Precompute projection derivatives
212  const float pt_prev_inv_z = 1.f / pt_prev.z;
213  const float pt_prev_inv_z_sq = pt_prev_inv_z * pt_prev_inv_z;
214 
215  const Vector3f d_proj_x(intrinsics_rgb.x * pt_prev_inv_z,
216  0.f,
217  -intrinsics_rgb.x * pt_prev.x * pt_prev_inv_z_sq);
218  const Vector3f d_proj_y(0.f,
219  intrinsics_rgb.y * pt_prev_inv_z,
220  -intrinsics_rgb.y * pt_prev.y * pt_prev_inv_z_sq);
221 
222  float nabla[6];
223 
224  const int numPara = shortIteration ? 3 : 6;
225  for (int para = 0; para < numPara; para++)
226  {
227  // Derivatives of approxInvPose wrt. the current parameter
228  Vector3f d_point_col;
229  switch (para + (shortIteration && !rotationOnly ? 3 : 0))
230  {
231  case 0: //rx
232  d_point_col = Vector3f(0, -pt_world.z, pt_world.y);
233  break;
234  case 1: // ry
235  d_point_col = Vector3f(pt_world.z, 0, -pt_world.x);
236  break;
237  case 2: // rz
238  d_point_col = Vector3f(-pt_world.y, pt_world.x, 0);
239  break; //rz
240  case 3: //tx
241  // Identity matrix
242  // We negate it because the ApplyDelta function uses the KinectFusion
243  // skew symmetric matrix and that matrix has negated rotation components.
244  // In order to use the rgb tracker we would need to negate the entire computed step, but given
245  // the peculiar structure of the increment matrix we only need to negate the translation component.
246  d_point_col = -Vector3f(1,0,0);
247  break;
248  case 4: //ty
249  d_point_col = -Vector3f(0,1,0);
250  break;
251  case 5: //tz
252  d_point_col = -Vector3f(0,0,1);
253  break;
254  default:
255  d_point_col = Vector3f(0,0,0); // Should never happen
256  break;
257  };
258 
259  // Chain the above with scenePose
260  const Vector3f d_point_col_rot(dot(scene_rot_row_0, d_point_col),
261  dot(scene_rot_row_1, d_point_col),
262  dot(scene_rot_row_2, d_point_col));
263 
264  // Chain with the projection
265  const Vector2f d_proj_point(dot(d_proj_x, d_point_col_rot),
266  dot(d_proj_y, d_point_col_rot));
267 
268  // Chain with the intensity gradient
269  nabla[para] = dot(gradient_prev, d_proj_point);
270  }
271 
272  // Weigh less the points far away from the camera
273  float depthWeight = 1.0f - (pt_curr.z - viewFrustum_min) / (viewFrustum_max - viewFrustum_min);
274  depthWeight = MAX(depthWeight, 0.f);
275  depthWeight *= depthWeight;
276 
277  // Compute the residual
278  localResidual = depthWeight * tukey_rho(intensity_diff, colourThresh);
279 
280  // Precompute tukey weights
281  const float tukey_coeff_gradient = depthWeight * tukey_rho_deriv(intensity_diff, colourThresh);
282  const float tukey_coeff_hessian = depthWeight * tukey_rho_deriv2(intensity_diff, colourThresh);
283 
284  // Fill gradient and hessian
285  for (int para = 0, counter = 0; para < numPara; para++)
286  {
287  localGradient[para] = tukey_coeff_gradient * nabla[para];
288 
289  for (int col = 0; col <= para; col++)
290  {
291  localHessian[counter++] = tukey_coeff_hessian * nabla[para] * nabla[col];
292  }
293  }
294 
295  return true;
296 }
