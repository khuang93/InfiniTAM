36 inline void calculate_projected_surfel_bounds(int locId, int indexImageWidth, int indexImageHeight, const ITMIntrinsics& intrinsics, float radius, float z,
37  int& cx, int& cy, int& projectedRadiusSquared, int& minX, int& minY, int& maxX, int& maxY)
38 {
39  // Calculate the (x,y) coordinates of the pixel in the index image to which the surfel's centre projects.
40  cx = locId % indexImageWidth, cy = locId / indexImageWidth;
41 
42  // Calculate the square of the radius of the circle to use to represent the projected surfel in the index image.
43  float f = 0.5f * (intrinsics.projectionParamsSimple.fx + intrinsics.projectionParamsSimple.fy);
44  int projectedRadius = static_cast<int>(radius * f / z + 0.5f);
45  projectedRadiusSquared = projectedRadius * projectedRadius;
46 
47  // Calculate the lower and upper bounds of a bounding box around the circle in the index image.
48  minX = cx - projectedRadius, maxX = cx + projectedRadius;
49  minY = cy - projectedRadius, maxY = cy + projectedRadius;
50 
51  // Clamp these bounds to the image bounds.
52  if(minX < 0) minX = 0;
53  if(maxX >= indexImageWidth) maxX = indexImageWidth - 1;
54  if(minY < 0) minY = 0;
55  if(maxY >= indexImageHeight) maxY = indexImageHeight - 1;
56 }

91 _CPU_AND_GPU_CODE_
92 inline bool project_surfel_to_index_image(const TSurfel& surfel, const Matrix4f& invT, const ITMIntrinsics& intrinsics,
93  int indexImageWidth, int indexImageHeight, int scaleFactor,
94  int& locId, float& z, int& scaledZ)
95 {
96  // Convert the surfel point into the coordinates of the current frame using v_i = T_i^{-1} v_i^g.
97  Vector3f p = surfel.position;
98  Vector4f vg(p.x, p.y, p.z, 1.0f);
99  Vector4f v = invT * vg;
100 
101  // If the point isn't in front of the viewer, early out.
102  z = v.z;
103  if(z <= 0.0f) return false;
104 
105  // Project the point onto the image plane of the current frame.
106  float ux = intrinsics.projectionParamsSimple.fx * v.x / z + intrinsics.projectionParamsSimple.px;
107  float uy = intrinsics.projectionParamsSimple.fy * v.y / z + intrinsics.projectionParamsSimple.py;
108 
109  // Convert the projected point into index image coordinates.
110  int x = static_cast<int>(ux * scaleFactor + 0.5f);
111  int y = static_cast<int>(uy * scaleFactor + 0.5f);
112 
113  // If the resulting point is outside the index image, early out.
114  if(x < 0 || x >= indexImageWidth || y < 0 || y >= indexImageHeight) return false;
115 
116  // Calculate the raster position of the point in the index image.
117  locId = y * indexImageWidth + x;
118 
119  // Calculate an integer depth value for the point for storage in the depth buffer.
120  scaledZ = static_cast<int>(z * 100000);
121 
122  return true;
123 }

351 _CPU_AND_GPU_CODE_
352 void shade_pixel_grey(int locId, const unsigned int *surfelIndexImage, const TSurfel *surfels, const Vector3f& lightPos, const Vector3f& viewerPos,
353  SurfelLightingType lightingType, Vector4u *outputImage)
354 {
355  const float ambient = lightingType == SLT_PHONG ? 0.3f : 0.2f;
356  const float lambertianCoefficient = lightingType == SLT_PHONG ? 0.35f : 0.8f;
357  const float phongCoefficient = 0.35f;
358  const float phongExponent = 20.0f;
359 
360  Vector4u value((uchar)0);
361 
362  int surfelIndex = surfelIndexImage[locId] - 1;
363  if(surfelIndex >= 0)
364  {
365  TSurfel surfel = surfels[surfelIndex];
366 
367  // Calculate the Lambertian lighting term.
368  Vector3f L = normalize(lightPos - surfel.position);
369  Vector3f N = surfel.normal;
370  float NdotL = ORUtils::dot(N, L);
371  float lambertian = CLAMP(NdotL, 0.0f, 1.0f);
372 
373  // Determine the intensity of the pixel using the Lambertian lighting equation.
374  float intensity = lightingType != SLT_FLAT ? ambient + lambertianCoefficient * lambertian : 1.0f;
375 
376  // If we're using Phong lighting:
377  if(lightingType == SLT_PHONG)
378  {
379  // Calculate the Phong lighting term.
380  Vector3f R = 2.0f * N * NdotL - L;
381  Vector3f V = normalize(viewerPos - surfel.position);
382  float phong = pow(CLAMP(dot(R,V), 0.0f, 1.0f), phongExponent);
383 
384  // Add the Phong lighting term to the intensity.
385  intensity += phongCoefficient * phong;
386  }
387 
388  // Fill in the final value for the pixel.
389  value = Vector4u((uchar)(intensity * 255.0f));
390  }
391 
392  outputImage[locId] = value;
393 }

439 _CPU_AND_GPU_CODE_TEMPLATE_
440 inline void update_depth_buffer_for_surfel(int surfelId, const TSurfel *surfels, const Matrix4f& invT, const ITMIntrinsics& intrinsics,
441  int indexImageWidth, int indexImageHeight, int scaleFactor, bool useRadii,
442  UnstableSurfelRenderingMode unstableSurfelRenderingMode, float stableSurfelConfidence,
443  int unstableSurfelZOffset, int *depthBuffer)
444 {
445  const TSurfel surfel = surfels[surfelId];
446 
447  // Check whether the surfel is unstable. If it is, and we're not rendering unstable surfels, early out.
448  bool unstableSurfel = surfel.confidence < stableSurfelConfidence;
449  if(unstableSurfel && unstableSurfelRenderingMode == USR_DONOTRENDER) return;
450 
451  // If the projection of the surfel falls within the bounds of the index image:
452  int locId, scaledZ;
453  float z;
454  if(project_surfel_to_index_image(surfel, invT, intrinsics, indexImageWidth, indexImageHeight, scaleFactor, locId, z, scaledZ))
455  {
456  // If the surfel's unstable and we're giving preference to stable surfels, add a z offset to ensure that
457  // it will only be rendered if there's no stable alternative along the same ray.
458  if(unstableSurfel && unstableSurfelRenderingMode == USR_FAUTEDEMIEUX) scaledZ += unstableSurfelZOffset;
459 
460  if(useRadii)
461  {
462  // If we're rendering the surfel as a circle rather than a point, calculate the radius of the projected
463  // surfel and its bounds within the index image.
464  int cx, cy, minX, minY, maxX, maxY, projectedRadiusSquared;
465  calculate_projected_surfel_bounds(
466  locId, indexImageWidth, indexImageHeight, intrinsics, surfel.radius, z,
467  cx, cy, projectedRadiusSquared, minX, minY, maxX, maxY
468  );
469 
470  // Rasterise the circle into the depth buffer.
471  for(int y = minY; y <= maxY; ++y)
472  {
473  int yOffset = y - cy;
474  int yOffsetSquared = yOffset * yOffset;
475 
476  for(int x = minX; x <= maxX; ++x)
477  {
478  int xOffset = x - cx;
479  int xOffsetSquared = xOffset * xOffset;
480  if(xOffsetSquared + yOffsetSquared > projectedRadiusSquared) continue;
481 
482  int offset = y * indexImageWidth + x;
483 
484 #if defined(__CUDACC__) && defined(__CUDA_ARCH__)
485  atomicMin(&depthBuffer[offset], scaledZ);
486 #else
487  // Note: No synchronisation is needed for the CPU version because it's not parallelised.
488  if(scaledZ < depthBuffer[offset]) depthBuffer[offset] = scaledZ;
489 #endif
490  }
491  }
492  }
493  else
494  {
495  // If we're rendering the surfel as a point, simply update the corresponding pixel in the depth buffer as necessary.
496 #if defined(__CUDACC__) && defined(__CUDA_ARCH__)
497  atomicMin(&depthBuffer[locId], scaledZ);
498 #else
499  // Note: No synchronisation is needed for the CPU version because it's not parallelised.
500  if(scaledZ < depthBuffer[locId]) depthBuffer[locId] = scaledZ;
501 #endif
502  }
503  }
504 }
