48  bool ProcessFrame(const ORUtils::Image<ElementType> *img, const ORUtils::SE3Pose *pose, int sceneId, int k, int nearestNeighbours[], float *distances, bool harvestKeyframes) const
49  {
50  // downsample and preprocess image => processedImage1
51  filterSubsample(img, processedImage1); // 320x240
52  filterSubsample(processedImage1, processedImage2); // 160x120
53  filterSubsample(processedImage2, processedImage1); // 80x60
54  filterSubsample(processedImage1, processedImage2); // 40x30
55 
56  filterGaussian(processedImage2, processedImage1, 2.5f);
57 
58  // compute code
59  int codeLength = encoding->getNumFerns();
60  char *code = new char[codeLength];
61  encoding->computeCode(processedImage1, code);
62 
63  // prepare outputs
64  int ret = -1;
65  bool releaseDistances = (distances == NULL);
66  if (distances == NULL) distances = new float[k];
67 
68  // find similar frames
69  int similarFound = relocDatabase->findMostSimilar(code, nearestNeighbours, distances, k);
70 
71  // add keyframe to database
72  if (harvestKeyframes)
73  {
74  if (similarFound == 0) ret = relocDatabase->addEntry(code);
75  else if (distances[0] > keyframeHarvestingThreshold) ret = relocDatabase->addEntry(code);
76 
77  if (ret >= 0) poseDatabase->storePose(ret, *pose, sceneId);
78  }
79 
80  // cleanup and return
81  delete[] code;
82  if (releaseDistances) delete[] distances;
83  return ret >= 0;
84  }
